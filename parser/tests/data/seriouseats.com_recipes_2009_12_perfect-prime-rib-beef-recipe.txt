function handleHeaderHamburgerClick(e) {

function toggleSection(e) {
  var clickedCategory = e.target.closest('.Nav__category');
  console.log('clicked cat', clickedCategory)
  var clickedIdx = clickedCategory.getAttribute('data-idx');
  var navEl = clickedCategory.closest('.Nav');
  var categories = navEl.querySelectorAll('.Nav__category');
  categories.forEach(function(category) {
      if (clickedIdx !== category.getAttribute('data-idx')) {
        category.classList.remove('Nav__category--open');
      } else {
        clickedCategory.classList.toggle('Nav__category--open');
      }
  });
};

<script>
$(document).ready(function(){
  if(window.innerWidth > 971) {
    var scrollIndicator,
        entryInView,
        $entryHeader,
        $entryBody,
        entryHeight,
        entryTop,
        globalNavHeight = seGlobalState["globalNavHeight"] || 0;

    // TO-DO: This is similar to finite scroll hasEntryChanged function - generalize and combine?
    function lastEntryInView(){
      // Returns jQuery obj of last div.content-main in view on page
      var lastEntry = $("div.content-main").last();
      // Iterate in reverse
      $($("div.content-main").get().reverse()).each(function() {
        var $thisEntry = $(this);
        var winBottom = window.pageYOffset + window.innerHeight,
            entryTop = $thisEntry.offset().top;
        if((winBottom - entryTop) > 0) {
          lastEntry = $thisEntry;
          return false;
        }
      });
      return lastEntry;
    }

    function setEntryInView() {
      scrollIndicator = $('.header-global progress');
      entryInView = lastEntryInView();
      $entryHeader = entryInView.find('.entry-header');
      $entryBody = entryInView.find('.entry-body, .recipe-body');
      if($entryHeader.length && $entryBody.length) {
        entryHeight = $entryHeader.height() + $entryBody.height();
        entryTop = $entryHeader.offset().top;
        scrollIndicator.attr('max', entryHeight);
      } else {
        entryTop = null;
      }
    }
    setEntryInView();

    if(scrollIndicator.length > 0 && typeof seGlobalState === "object" && typeof seGlobalState.currentY !== "undefined") {

      function scrollProgress() {
        if(entryTop !== null) {
          var scrollPercent = Math.round(seGlobalState.currentY + globalNavHeight - entryTop);
          if(scrollPercent >= 0) {
            scrollIndicator.attr('value', scrollPercent);
          } else {
            scrollIndicator.attr('value', entryHeight);
          }
        }
      }
      setInterval(function() {
        if(seGlobalState.didScroll) {
          setEntryInView();
        }
      // One second is probably frequent enough for normal scrolling - if we see a big lag, can decrease
      }, 1000);
      seGlobalState.$window.on("scroll", function() {
        scrollProgress();
        // setScrollVars();
      });
      seGlobalState.$window.on("load resize", function(){
        setEntryInView();
        scrollProgress();
      });

    } // End if page has scroll indicator bar
  } // End if desktop size

});

</script>

          <div id="entry-text" class="recipe-introduction">

            <div class="recipe-introduction-body">
              <p class="caption">[Photographs and video: J. Kenji López-Alt]</p>
<p>Prime rib is one of our favorite cuts of beef. It's also expensive, which means you want the best, most reliable results possible. This recipe takes advantage of <a href="https://www.seriouseats.com/2017/03/how-to-reverse-sear-best-way-to-cook-steak.html">the reverse sear method</a> to yield prime rib with a deep brown, crisp, crackly salty crust surrounding a tender, juicy, medium-rare interior. By starting the roast off in a lower temperature oven, we evenly cook the meat and reduce its surface moisture, so that a finishing blast of high heat quickly gives it a perfectly burnished exterior.</p>
<p></p>

            </div>

            <div class="recipe-why">
              <h3 class="title-section">Why It Works</h3>
              <div class="recipe-why-body"><ul><li>A low and slow start delivers perfectly evenly cooked medium-rare doneness all the way from edge to center.</li>
<li>Blasting the prime rib with heat just before serving gives you a crackling-crisp, browned crust.</li></ul></div>
            </div>

                <p><strong>Read more: <a href="https://www.seriouseats.com/2009/12/the-food-lab-how-to-cook-roast-a-perfect-prime-rib.html" data-click-category="recipe" data-click-id="read-more">The Food Lab: How to Cook a Perfect Prime Rib</a></strong></p>

                <h2 class="title-section">Directions</h2>
                <ol class="recipe-procedures-list instructions">
                    <li class="recipe-procedure">
                      <div class="recipe-procedure-number checked">1.</div>
                      <div class="recipe-procedure-text">
                        <p></p><p>Preheat oven to lowest possible temperature setting, 150°F (66°C) or higher if necessary. (Some ovens cannot hold a temperature below 250°F/121°C.) Season roast generously with salt and pepper. Place roast, with fat cap up, on a V-rack set in a large roasting pan, or on a wire rack set in a rimmed baking sheet. Place in oven and cook until center of roast registers 120-125°F (49-52°C) on an instant-read thermometer for rare, 130°F (54°C)  for medium-rare, or 135°F (57°C) for medium to medium-well. In a 150°F oven, this will take around 5 1/2 to 6 1/2 hours; in a 250°F oven, this will take 3 1/2 to 4 hours.</p><p></p>
                      </div>
                        <figure class="recipe-procedure-image hide">
                          <img src="https://static.seriouseats.com/1/braestar/live/img/placeholder-entry-image.png" data-src="https://www.seriouseats.com/recipes/20151202-prime-rib-red-wine-jus-step-by-step-kenji-15-750x563.jpg">

                    </li>
                    <li class="recipe-procedure">
                      <div class="recipe-procedure-number ">2.</div>
                      <div class="recipe-procedure-text">
                        <p></p><p>Remove roast from oven and tent loosely with aluminum foil. Place in a warm spot in the kitchen and allow to rest for at least 30 minutes and up to 1 1/2 hours. Meanwhile, preheat oven to highest possible temperature setting, 500 to 550°F (260 to 288°C).</p><p></p>
                      </div>
                    </li>
                    <li class="recipe-procedure">
                      <div class="recipe-procedure-number ">3.</div>
                      <div class="recipe-procedure-text">
                        <p></p><p>Ten minutes before guests are ready to be served, remove foil, place roast back in hot oven, and cook until well browned and crisp on the exterior, 6 to 10 minutes. Remove from oven, carve, and serve immediately.</p><p></p>

              <aside class="callout callout-bottom callout-bottom-recipe recipe-notes"><h6 class="callout-title">Notes</h6> <span class="info"><p>This recipe works for prime rib roasts of any size from two ribs to six ribs. Plan on one pound of bone-in roast per guest. (Each rib adds one and a half to two pounds to the roast.) For best results, use a dry-aged prime-grade or grass-fed roast.</p>

<p>To improve the crust, allow the roast to air-dry, uncovered, on a rack in the refrigerator overnight before roasting. Seasoning with salt up to a day in advance will help the seasoning penetrate the meat more deeply. If, after step 1, your timing is off, and your roast is ready long before your guests are, reheat the roast by placing it in a 200°F (93°C) oven for 45 minutes before you continue with step 2.</p></span></aside>

          var nextEntry = $(allEntries[i+1]),
              nextEntryID = nextEntry.data("id") || null;

          if(nextEntryID !== null && nextEntryID != obj.currentEntryID) {
            // Update current post/history
            obj.updateCurrentEntry(nextEntryID, nextEntry.data("title"), nextEntry.data("permalink"));
          }

        }

        // Else, the mid-posts break is in the middle of the page, and we don't update the history
        // break loop
        return false;

      });
      
    }, // hasEntryChanged

    loadNextEntry: function() {
      var obj = this,
          winBottom = window.pageYOffset + window.innerHeight,
          lastEntryBottom = obj.lastEntry.offset().top + obj.lastEntry.outerHeight(true);

      if(lastEntryBottom - obj.loadThreshold > winBottom) {

        obj.loadTry = false;
        setTimeout(function(){ obj.fsTOInit(); }, obj.checkTime);

      } else {

        var url = obj.nextUrl.pop();

        if(!url) {return;}

        obj.counter++;

        var requestUrl = "/json/entry?record=" + url + "&counter=" + obj.counter;

          url: requestUrl

              if (!data.hasOwnProperty("body")) {
                return;

              dataLayer.push({
                "event": "gaEvent",
                "category": "finite-scroll",
                "action": "load-next-entry",
                "label": obj.counter
              });
            } catch (err) {}            

            try {
              var $relish_button = $("span#relish-button[data-recipe-id='" + data.post_id + "']");
              if ($relish_button.length > 0)
              {
                if (window.fexyRelish === undefined)
                {
                  $.ajax({
                    url: "https://platform.relish.com/scripts/relish.js?key=fad43e07-2ce3-4c04-97a2-ffd099ecbfd9",
                    dataType: "script",
                    cache: true,
                    success: function() {
                      window.fexyRelish.addShopButton($relish_button, null, data.post_id);

            catch (err) {}

            seGlobalState.pinImagesLoaded = false;

          } catch (err) {}

          obj.lastEntry = obj.getLastEntry();

          if(obj.nextUrl.length) {
            obj.loadTry = false;
            setTimeout(function(){ obj.fsTOInit(); }, obj.checkTime);
          }
        }); // ajax done
      }
    }, // loadNextEntry

    fsTOInit: function() {
      var obj = this;
      if(obj.loadTry) {
        obj.loadNextEntry();
      } else {
        setTimeout(function(){ obj.fsTOInit(); }, obj.checkTime);
      }
    },

        obj.hasEntryChanged();

        setTimeout(function() {
          obj.historyTry = false;

    } // setListeners
